# Docker Compose

# Docker compose file

docker-compose.yml

web:
    build: .
    links:
        - redis
    ports:
        - "80:5000"
redis:
    image: redis

# Format name

<project>_<service>_<number>

# Commands

# Build image

$ docker-compose build

# Container

  • run
$ docker-compose up
  • -d refresh containers due to changes
$ docker-compose up -d
  • logs
$ docker-compose logs <SERVICE>
  • stop all containers
$ docker-compose stop
  • remove stoped containers
$ docker-compose rm <SERVICE>
  • remove all containers
$ docker-compose rm
  • stop and remove
$ docker-compose down

# Rename docker-compose file

$ docker-compose -f file_name.yml

# Scale

# Load balancer

docker-compose.yml

web:
    build: .
    links:
        - redis
    ports:
        - "80:5000"
redis:
    image: redis
lb:
    image: tutum/haproxy
    links: 
        - web
    ports:
        - "80:80"
    environment:
        - BACKEND_PORT=5000
        - BALANCE=roundrobin

# Run

docker-compose scale web=10

# Recreate without dependencies

docker-compose build --force-recreate --no-deps lb
